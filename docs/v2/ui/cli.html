 54    def _setup_players(self) -> None:
 55        """设置玩家.
 56        
 57        为游戏添加指定数量的玩家，包括人类玩家和AI玩家。
 58        """
 59        # 添加玩家到游戏状态
 60        for i in range(self.num_players):
 61            if i == self.human_seat:
 62                name = "You"
 63            else:
 64                name = f"AI_{i}"
 65            
 66            # 通过控制器的游戏状态添加玩家
 67            snapshot = self.controller.get_snapshot()
 68            if len(snapshot.players) <= i:
 69                # 需要添加玩家
 70                player = Player(
 71                    seat_id=i,
 72                    name=name,
 73                    chips=self.initial_chips
 74                )
 75                self.controller._game_state.add_player(player)

 77    def run(self) -> None:
 78        """运行游戏主循环.
 79        
 80        执行完整的德州扑克游戏流程，包括多手牌的连续游戏。
 81        """
 82        # 显示游戏开始信息
 83        header = self.renderer.render_game_header(
 84            0, self.num_players, self.initial_chips, self.human_seat
 85        )
 86        self.logger.info(header)
 87        
 88        hand_count = 0
 89        
 90        while True:
 91            hand_count += 1
 92            
 93            # 显示手牌开始信息
 94            hand_header = self.renderer.render_game_header(
 95                hand_count, self.num_players, self.initial_chips, self.human_seat
 96            )
 97            self.logger.info(f"\n{hand_header.split('===')[-1].strip()}")
 98            
 99            # 开始新手牌
100            if not self.controller.start_new_hand():
101                game_over_msg = self.renderer.render_game_over("活跃玩家不足")
102                self.logger.info(game_over_msg)
103                break
104            
105            # 显示初始状态
106            self._display_game_state()
107            
108            # 游戏循环
109            while not self.controller.is_hand_over():
110                current_player_id = self.controller.get_current_player_id()
111                
112                if current_player_id is None:
113                    break
114                
115                if current_player_id == self.human_seat:
116                    # 人类玩家行动
117                    self._handle_human_action()
118                else:
119                    # AI玩家行动
120                    self._handle_ai_action(current_player_id)
121                
122                # 显示更新后的状态
123                self._display_game_state()
124            
125            # 结束手牌
126            result = self.controller.end_hand()
127            if result:
128                self._display_hand_result(result)
129            
130            # 检查是否继续游戏
131            if not self._should_continue():
132                break
133        
134        self.logger.info("\n感谢游戏！")

136    def _display_game_state(self) -> None:
137        """显示当前游戏状态.
138        
139        使用渲染器将游戏状态快照转换为格式化的显示内容。
140        """
141        snapshot = self.controller.get_snapshot()
142        state_display = self.renderer.render_game_state(snapshot, self.human_seat)
143        self.logger.info(f"\n{state_display}")

145    def _handle_human_action(self) -> None:
146        """处理人类玩家行动.
147        
148        使用输入处理器获取用户输入，并通过控制器执行行动。
149        """
150        try:
151            snapshot = self.controller.get_snapshot()
152            action_input = self.input_handler.get_player_action(snapshot, self.human_seat)
153            
154            # 转换为Action对象
155            action = Action(
156                player_id=action_input.player_id,
157                action_type=action_input.action_type,
158                amount=action_input.amount
159            )
160            
161            # 执行行动
162            success = self.controller.execute_action(action)
163            if not success:
164                error_msg = self.renderer.render_error_message("行动执行失败")
165                self.logger.info(error_msg)
166                
167        except Exception as e:
168            error_msg = self.renderer.render_error_message(str(e))
169            self.logger.info(error_msg)
170    
171    def _handle_ai_action(self, player_id: int) -> None:
172        """处理AI玩家行动.
173        
174        Args:
175            player_id: AI玩家ID
176        """
177        snapshot = self.controller.get_snapshot()
178        player = next(p for p in snapshot.players if p.seat_id == player_id)
179        
180        try:
181            # 让控制器处理AI行动
182            success = self.controller.process_ai_action()
183            
184            if success:
185                # 获取更新后的快照来显示AI行动
186                new_snapshot = self.controller.get_snapshot()
187                # 这里可以通过事件系统获取具体的行动描述
188                # 暂时使用简单的显示
189                ai_action_msg = self.renderer.render_ai_action(
190                    player.name, "执行了行动"
191                )
192                self.logger.info(ai_action_msg)
193            else:
194                error_msg = self.renderer.render_error_message(f"{player.name} AI行动失败")
195                self.logger.info(error_msg)
196                
197        except Exception as e:
198            error_msg = self.renderer.render_error_message(f"{player.name} AI行动异常: {e}")
199            self.logger.info(error_msg)

201    def _display_hand_result(self, result: HandResult) -> None:
202        """显示手牌结果.
203        
204        Args:
205            result: 手牌结果对象
206        """
207        snapshot = self.controller.get_snapshot()
208        result_display = self.renderer.render_hand_result(result, snapshot)
209        self.logger.info(result_display)

211    def _should_continue(self) -> bool:
212        """检查是否应该继续游戏.
213        
214        Returns:
215            True表示继续，False表示退出
216        """
217        # 检查是否还有足够的活跃玩家
218        snapshot = self.controller.get_snapshot()
219        active_players = [
220            p for p in snapshot.players 
221            if p.status == SeatStatus.ACTIVE and p.chips > 0
222        ]
223        
224        if len(active_players) < 2:
225            game_over_msg = self.renderer.render_game_over("只剩一个玩家有筹码")
226            self.logger.info(game_over_msg)
227            return False
228        
229        # 询问用户是否继续
230        return self.input_handler.get_continue_choice()

 20    @staticmethod
 21    def render_game_header(hand_count: int, num_players: int, initial_chips: int, human_seat: int) -> str:
 22        """渲染游戏头部信息.
 23        
 24        Args:
 25            hand_count: 手牌计数
 26            num_players: 玩家数量
 27            initial_chips: 初始筹码
 28            human_seat: 人类玩家座位号
 29            
 30        Returns:
 31            格式化的头部信息字符串
 32        """
 33        lines = [
 34            "=== 德州扑克 v2 CLI ===",
 35            f"玩家数: {num_players}, 初始筹码: {initial_chips}",
 36            f"您是玩家 {human_seat}",
 37            "",
 38            f"=== 第 {hand_count} 手牌 ==="
 39        ]
 40        return "\n".join(lines)

 42    @staticmethod
 43    def render_game_state(snapshot: GameSnapshot, human_seat: int) -> str:
 44        """渲染当前游戏状态.
 45        
 46        Args:
 47            snapshot: 游戏状态快照
 48            human_seat: 人类玩家座位号
 49            
 50        Returns:
 51            格式化的游戏状态字符串
 52        """
 53        lines = []
 54        
 55        # 基本信息
 56        lines.append(f"阶段: {snapshot.phase.name}")
 57        lines.append(f"底池: {snapshot.pot}")
 58        lines.append(f"当前最高下注: {snapshot.current_bet}")
 59        
 60        # 公共牌
 61        if snapshot.community_cards:
 62            cards_str = " ".join([CLIRenderer._format_card(card) 
 63                                for card in snapshot.community_cards])
 64            lines.append(f"公共牌: {cards_str}")
 65        
 66        # 玩家状态
 67        lines.append("")
 68        lines.append("玩家状态:")
 69        for player in snapshot.players:
 70            player_line = CLIRenderer._render_player_status(
 71                player, snapshot.current_player, human_seat
 72            )
 73            lines.append(f"  {player_line}")
 74        
 75        return "\n".join(lines)

 78    def render_action_prompt(player_name: str, chips: int, available_actions: List[tuple]) -> str:
 79        """渲染行动提示.
 80        
 81        Args:
 82            player_name: 玩家名称
 83            chips: 玩家筹码

 92        ]

100    def render_hand_result(result: HandResult, snapshot: GameSnapshot) -> str:
101        """渲染手牌结果.
102        
103        Args:
104            result: 手牌结果
105            snapshot: 最终游戏状态快照

134                                if p.seat_id in side_pot.eligible_players]

136

132    @staticmethod
133    def _parse_text_command(
134        command: str, 
135        available_actions: List[Tuple[ActionType, str, int]], 
136        player_id: int,
137        snapshot: GameSnapshot,
138        player
139    ) -> Optional[ActionInput]:
140        """解析文本命令.
141        
142        Args:
143            command: 用户输入的文本命令
144            available_actions: 可用行动列表
145            player_id: 玩家ID
146            snapshot: 游戏状态快照
147            player: 玩家对象
148            
149        Returns:
150            解析后的ActionInput对象，如果解析失败返回None
151        """
152        command = command.lower().strip()
153        
154        # 创建行动类型映射
155        action_map = {action_type: (action_type, amount) for action_type, _, amount in available_actions}
156        
157        # 解析常见命令
158        if command in ['fold', 'f', '弃牌']:
159            if ActionType.FOLD in action_map:
160                action_type, amount = action_map[ActionType.FOLD]
161                return ActionInput(player_id=player_id, action_type=action_type, amount=amount)
162        
163        elif command in ['check', 'c', '过牌']:
164            if ActionType.CHECK in action_map:
165                action_type, amount = action_map[ActionType.CHECK]
166                return ActionInput(player_id=player_id, action_type=action_type, amount=amount)
167        
168        elif command in ['call', '跟注']:
169            if ActionType.CALL in action_map:
170                action_type, amount = action_map[ActionType.CALL]
171                return ActionInput(player_id=player_id, action_type=action_type, amount=amount)
172        
173        elif command in ['allin', 'all', '全押']:
174            if ActionType.ALL_IN in action_map:
175                action_type, amount = action_map[ActionType.ALL_IN]
176                return ActionInput(player_id=player_id, action_type=action_type, amount=amount)
177        
178        # 解析带金额的命令
179        elif command.startswith('bet ') or command.startswith('下注 '):
180            if ActionType.BET in action_map:
181                try:
182                    amount_str = command.split(' ', 1)[1]
183                    amount = int(amount_str)
184                    return ActionInput(player_id=player_id, action_type=ActionType.BET, amount=amount)
185                except (ValueError, IndexError):
186                    pass
187        
188        elif command.startswith('raise ') or command.startswith('加注 '):
189            if ActionType.RAISE in action_map:
190                try:
191                    amount_str = command.split(' ', 1)[1]
192                    amount = int(amount_str)
193                    return ActionInput(player_id=player_id, action_type=ActionType.RAISE, amount=amount)
194                except (ValueError, IndexError):
195                    pass
196        
197        return None

245    @staticmethod
246    def _get_available_actions(player, snapshot: GameSnapshot) -> List[Tuple[ActionType, str, int]]:
247        """获取玩家可用行动列表.
248        
249        Args:
250            player: 玩家快照对象
251            snapshot: 游戏状态快照
252            
253        Returns:
254            可用行动列表 [(action_type, description, amount), ...]
255        """
256        actions = []
257        
258        # 弃牌 - 总是可用
259        actions.append((ActionType.FOLD, "弃牌", 0))
260        
261        # 检查当前下注情况
262        call_amount = snapshot.current_bet - player.current_bet
263        
264        if call_amount == 0:
265            # 没有需要跟注的金额
266            actions.append((ActionType.CHECK, "过牌", 0))
267            
268            # 可以下注
269            if player.chips > 0:
270                min_bet = snapshot.big_blind if hasattr(snapshot, 'big_blind') else 10
271                if player.chips >= min_bet:
272                    actions.append((ActionType.BET, f"下注 (最少 {min_bet})", min_bet))
273                
274                # 全押
275                actions.append((ActionType.ALL_IN, f"全押 ({player.chips})", player.chips))
276        else:
277            # 需要跟注
278            if player.chips >= call_amount:
279                actions.append((ActionType.CALL, f"跟注 ({call_amount})", call_amount))
280                
281                # 可以加注
282                min_raise = call_amount + (snapshot.last_raise_amount or snapshot.big_blind if hasattr(snapshot, 'big_blind') else 10)
283                if player.chips > call_amount and player.chips >= min_raise:
284                    actions.append((ActionType.RAISE, f"加注 (最少到 {min_raise})", min_raise))
285            
286            # 全押
287            if player.chips > 0:
288                actions.append((ActionType.ALL_IN, f"全押 ({player.chips})", player.chips))
289        
290        return actions

293    def _get_bet_amount(action_type: ActionType, snapshot: GameSnapshot, player) -> int:
294        """获取下注/加注的具体金额.
295        
296        Args:
297            action_type: 行动类型
298            snapshot: 游戏状态快照
299            player: 玩家快照对象
300            
301        Returns:
302            验证后的下注金额
303            
304        Raises:
305            click.Abort: 用户取消输入
306        """
307        if action_type == ActionType.BET:
308            min_amount = snapshot.big_blind if hasattr(snapshot, 'big_blind') else 10
309            max_amount = player.chips
310            action_name = "下注"
311        elif action_type == ActionType.RAISE:
312            call_amount = snapshot.current_bet - player.current_bet
313            min_raise_amount = snapshot.last_raise_amount or (snapshot.big_blind if hasattr(snapshot, 'big_blind') else 10)
314            min_amount = snapshot.current_bet + min_raise_amount
315            max_amount = player.chips + player.current_bet
316            action_name = "加注到"
317        else:
318            raise ValueError(f"不支持的行动类型: {action_type}")
319        
320        # 确保最小金额不超过玩家筹码
321        min_amount = min(min_amount, max_amount)
322        
323        while True:
324            try:
325                amount = CLIInputHandler.get_bet_amount_input(
326                    min_amount, max_amount, action_name
327                )
328                
329                # 对于加注，返回总下注金额
330                if action_type == ActionType.RAISE:
331                    return amount
332                else:
333                    return amount
334                    
335            except click.Abort:
336                # 用户取消，询问是否改为其他行动
337                if click.confirm("取消下注，是否改为过牌/跟注?", default=True):
338                    if snapshot.current_bet == player.current_bet:
339                        return 0  # CHECK
340                    else:
341                        return snapshot.current_bet - player.current_bet  # CALL
342                else:
343                    raise
344            except Exception as e:
345                click.echo(f"输入错误: {e}")
346                click.echo("请重新输入")
