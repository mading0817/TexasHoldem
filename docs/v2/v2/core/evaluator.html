 16@dataclass(frozen=True)
 17class HandResult:
 18    """
 19    牌型评估结果。
 20    
 21    包含牌型等级和用于比较的关键牌值。
 22    
 23    Attributes:
 24        rank: 牌型等级
 25        primary_value: 主要牌值（如对子的点数）
 26        secondary_value: 次要牌值（如两对中较小的对子）
 27        kickers: 踢脚牌（按大小降序排列）
 28    """
 29    rank: HandRank
 30    primary_value: int
 31    secondary_value: int = 0
 32    kickers: Tuple[int, ...] = ()
 33
 34    def __post_init__(self):
 35        """验证结果数据的有效性。
 36        
 37        Raises:
 38            ValueError: 如果牌型等级或牌值无效
 39        """
 40        if not isinstance(self.rank, HandRank):
 41            raise ValueError(f"无效的牌型等级: {self.rank}")
 42        
 43        if self.primary_value < 2 or self.primary_value > 14:
 44            raise ValueError(f"无效的主要牌值: {self.primary_value}")
 45
 46    def compare_to(self, other: 'HandResult') -> int:
 47        """
 48        比较两个牌型结果。
 49        
 50        Args:
 51            other: 另一个牌型结果
 52            
 53        Returns:
 54            1 if self > other, -1 if self < other, 0 if equal
 55        """
 56        # 首先比较牌型等级
 57        if self.rank != other.rank:
 58            return 1 if self.rank > other.rank else -1
 59        
 60        # 牌型相同，比较主要牌值
 61        if self.primary_value != other.primary_value:
 62            return 1 if self.primary_value > other.primary_value else -1
 63        
 64        # 主要牌值相同，比较次要牌值
 65        if self.secondary_value != other.secondary_value:
 66            return 1 if self.secondary_value > other.secondary_value else -1
 67        
 68        # 比较踢脚牌
 69        for my_kicker, other_kicker in zip(self.kickers, other.kickers):
 70            if my_kicker != other_kicker:
 71                return 1 if my_kicker > other_kicker else -1
 72        
 73        # 完全相同
 74        return 0

 76    def __str__(self) -> str:
 77        """返回牌型的可读描述。
 78        
 79        Returns:
 80            牌型的中文描述字符串
 81        """
 82        rank_names = {
 83            HandRank.HIGH_CARD: "高牌",
 84            HandRank.ONE_PAIR: "一对",
 85            HandRank.TWO_PAIR: "两对",
 86            HandRank.THREE_OF_A_KIND: "三条",
 87            HandRank.STRAIGHT: "顺子",
 88            HandRank.FLUSH: "同花",
 89            HandRank.FULL_HOUSE: "葫芦",
 90            HandRank.FOUR_OF_A_KIND: "四条",
 91            HandRank.STRAIGHT_FLUSH: "同花顺",
 92            HandRank.ROYAL_FLUSH: "皇家同花顺"
 93        }
 94        
 95        rank_name = rank_names.get(self.rank, str(self.rank))
 96        
 97        if self.rank == HandRank.ONE_PAIR:
 98            return f"{rank_name}({Rank(self.primary_value).name})"
 99        elif self.rank == HandRank.TWO_PAIR:
100            return f"{rank_name}({Rank(self.primary_value).name}和{Rank(self.secondary_value).name})"
101        elif self.rank in [HandRank.THREE_OF_A_KIND, HandRank.FOUR_OF_A_KIND]:
102            return f"{rank_name}({Rank(self.primary_value).name})"
103        elif self.rank == HandRank.FULL_HOUSE:
104            return f"{rank_name}({Rank(self.primary_value).name}带{Rank(self.secondary_value).name})"
105        elif self.rank in [HandRank.STRAIGHT, HandRank.STRAIGHT_FLUSH, HandRank.ROYAL_FLUSH]:
106            return f"{rank_name}({Rank(self.primary_value).name}高)"
107        else:
108            return rank_name

112    """
113    简单牌型评估器。

118    def evaluate_hand(self, hole_cards: List[Card], community_cards: List[Card]) -> HandResult:
119        """
120        评估7张牌中的最佳5张牌组合。
121        
122        Args:
123            hole_cards: 玩家手牌（2张）
124            community_cards: 公共牌（最多5张）
125            
126        Returns:
127            最佳牌型结果
128            
129        Raises:
130            ValueError: 如果输入牌数不符合要求
131        """
132        if len(hole_cards) != 2:
133            raise ValueError(f"手牌必须是2张，实际: {len(hole_cards)}")
134        
135        if len(community_cards) > 5:
136            raise ValueError(f"公共牌不能超过5张，实际: {len(community_cards)}")
137        
138        all_cards = hole_cards + community_cards
139        if len(all_cards) < 5:
140            raise ValueError(f"总牌数不足5张，无法评估: {len(all_cards)}")
141        
142        # 从所有牌中找出最佳5张牌组合
143        return self._find_best_hand(all_cards)

145    def _find_best_hand(self, cards: List[Card]) -> HandResult:
146        """
147        从给定牌中找出最佳5张牌组合。
148        
149        Args:
150            cards: 所有可用的牌
151            
152        Returns:
153            最佳牌型结果
154        """
155        if len(cards) == 5:
156            return self._evaluate_five_cards(cards)
157        
158        # 如果超过5张牌，尝试所有5张牌的组合
159        best_result = None
160        for five_cards in combinations(cards, 5):
161            result = self._evaluate_five_cards(list(five_cards))
162            if best_result is None or result.compare_to(best_result) > 0:
163                best_result = result
164        
165        return best_result

167    def _evaluate_five_cards(self, cards: List[Card]) -> HandResult:
168        """
169        评估恰好5张牌的牌型。
170        
171        Args:
172            cards: 恰好5张牌
173            
174        Returns:
175            牌型结果
176            
177        Raises:
178            ValueError: 如果不是恰好5张牌
179        """
180        if len(cards) != 5:
181            raise ValueError(f"必须是5张牌，实际: {len(cards)}")
182        
183        # 按点数降序排序
184        sorted_cards = sorted(cards, key=lambda c: c.rank.value, reverse=True)
185        ranks = [card.rank.value for card in sorted_cards]
186        suits = [card.suit for card in sorted_cards]
187        
188        # 统计点数出现次数
189        rank_counts = CollectionsCounter(ranks)
190        count_groups = CollectionsCounter(rank_counts.values())
191        
192        # 检查是否为同花
193        is_flush = len(set(suits)) == 1
194        
195        # 检查是否为顺子
196        is_straight, straight_high = self._check_straight(ranks)
197        
198        # 按优先级检查各种牌型
199        if is_straight and is_flush:
200            # 检查是否为皇家同花顺
201            if straight_high == 14 and set(ranks) == {14, 13, 12, 11, 10}:
202                return HandResult(HandRank.ROYAL_FLUSH, straight_high)
203            return HandResult(HandRank.STRAIGHT_FLUSH, straight_high)
204        
205        if count_groups[4] == 1:  # 四条
206            four_rank = self._get_rank_with_count(rank_counts, 4)
207            kicker = self._get_rank_with_count(rank_counts, 1)
208            return HandResult(HandRank.FOUR_OF_A_KIND, four_rank, kickers=(kicker,))
209        
210        if count_groups[3] == 1 and count_groups[2] == 1:  # 葫芦
211            three_rank = self._get_rank_with_count(rank_counts, 3)
212            pair_rank = self._get_rank_with_count(rank_counts, 2)
213            return HandResult(HandRank.FULL_HOUSE, three_rank, pair_rank)
214        
215        if is_flush:
216            return HandResult(HandRank.FLUSH, ranks[0], kickers=tuple(ranks[1:]))
217        
218        if is_straight:
219            return HandResult(HandRank.STRAIGHT, straight_high)
220        
221        if count_groups[3] == 1:  # 三条
222            three_rank = self._get_rank_with_count(rank_counts, 3)
223            kickers = tuple(sorted([r for r in ranks if r != three_rank], reverse=True))
224            return HandResult(HandRank.THREE_OF_A_KIND, three_rank, kickers=kickers)
225        
226        if count_groups[2] == 2:  # 两对
227            pairs = sorted([r for r, c in rank_counts.items() if c == 2], reverse=True)
228            kicker = self._get_rank_with_count(rank_counts, 1)
229            return HandResult(HandRank.TWO_PAIR, pairs[0], pairs[1], kickers=(kicker,))
230        
231        if count_groups[2] == 1:  # 一对
232            pair_rank = self._get_rank_with_count(rank_counts, 2)
233            kickers = tuple(sorted([r for r in ranks if r != pair_rank], reverse=True))
234            return HandResult(HandRank.ONE_PAIR, pair_rank, kickers=kickers)
235        
236        # 高牌
237        return HandResult(HandRank.HIGH_CARD, ranks[0], kickers=tuple(ranks[1:]))

239    def _check_straight(self, ranks: List[int]) -> Tuple[bool, int]:
240        """
241        检查是否为顺子。
242        
243        Args:
244            ranks: 按降序排列的点数列表
245            
246        Returns:
247            (是否为顺子, 顺子的最高牌)
248        """
249        # 去重并排序
250        unique_ranks = sorted(set(ranks), reverse=True)
251        
252        if len(unique_ranks) < 5:
253            return False, 0
254        
255        # 检查普通顺子
256        for i in range(len(unique_ranks) - 4):
257            if unique_ranks[i] - unique_ranks[i + 4] == 4:
258                return True, unique_ranks[i]
259        
260        # 检查A-2-3-4-5的特殊顺子
261        if set(unique_ranks[:5]) == {14, 5, 4, 3, 2}:
262            return True, 5  # A-2-3-4-5顺子以5为高牌
263        
264        return False, 0

266    def _get_rank_with_count(self, rank_counts: Counter, count: int) -> int:
267        """
268        获取出现指定次数的点数。
269        
270        Args:
271            rank_counts: 点数计数器
272            count: 目标出现次数
273            
274        Returns:
275            符合条件的点数
276            
277        Raises:
278            ValueError: 如果未找到符合条件的点数
279        """
280        for rank, cnt in rank_counts.items():
281            if cnt == count:
282                return rank
283        raise ValueError(f"未找到出现{count}次的点数")
