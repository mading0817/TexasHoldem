 62class ActionValidator:
 63    """Action validator for poker player actions.
 64    
 65    Validates player actions and provides intelligent conversion of invalid
 66    actions to valid alternatives when possible.

 69    def validate(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
 70        """Validate and potentially convert a player action.
 71        
 72        Args:
 73            game_state: The current game state.
 74            player: The player attempting the action.
 75            action: The action to validate.
 76            
 77        Returns:
 78            A ValidatedAction containing the original action and validated parameters.
 79            
 80        Raises:
 81            InvalidActionError: When the action is invalid and cannot be converted.
 82            InsufficientChipsError: When the player lacks chips for the action.
 83        """
 84        # Basic validation
 85        self._validate_basic_conditions(game_state, player, action)
 86        
 87        # Validate specific action types
 88        if action.action_type == ActionType.FOLD:
 89            return self._validate_fold(player, action)
 90        elif action.action_type == ActionType.CHECK:
 91            return self._validate_check(game_state, player, action)
 92        elif action.action_type == ActionType.CALL:
 93            return self._validate_call(game_state, player, action)
 94        elif action.action_type == ActionType.BET:
 95            return self._validate_bet(game_state, player, action)
 96        elif action.action_type == ActionType.RAISE:
 97            return self._validate_raise(game_state, player, action)
 98        elif action.action_type == ActionType.ALL_IN:
 99            return self._validate_all_in(player, action)
100        else:
101            raise InvalidActionError(f"Unknown action type: {action.action_type}")

103    def _validate_basic_conditions(self, game_state: GameStateProtocol, player: Player, action: Action) -> None:
104        """Validate basic conditions for any action.
105        
106        Args:
107            game_state: The current game state.
108            player: The player attempting the action.
109            action: The action to validate.
110            
111        Raises:
112            InvalidActionError: When basic conditions are not met.
113        """
114        # Check if player can act
115        if player.status != SeatStatus.ACTIVE:
116            raise InvalidActionError(f"Player {player.seat_id} cannot act, status: {player.status.value}")
117        
118        # Check if it's the player's turn
119        if game_state.current_player_seat is None:
120            raise InvalidActionError("No player can act, game phase should transition")
121        
122        if game_state.current_player_seat != player.seat_id:
123            raise InvalidActionError(
124                f"Not player {player.seat_id}'s turn, current player: {game_state.current_player_seat}"
125            )
126        
127        # Validate action amount for betting actions
128        if action.action_type in [ActionType.BET, ActionType.RAISE]:
129            if action.amount <= 0:
130                raise InvalidActionError(f"Bet/raise amount must be positive: {action.amount}")

132    def _validate_fold(self, player: Player, action: Action) -> ValidatedAction:
133        """Validate a fold action.
134        
135        Args:
136            player: The player attempting to fold.
137            action: The fold action.
138            
139        Returns:
140            A validated fold action.
141        """
142        return ValidatedAction(
143            original_action=action,
144            actual_action_type=ActionType.FOLD,
145            actual_amount=0,
146            player_seat=player.seat_id
147        )

149    def _validate_check(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
150        """Validate a check action.
151        
152        Args:
153            game_state: The current game state.
154            player: The player attempting to check.
155            action: The check action.
156            
157        Returns:
158            A validated check action.
159            
160        Raises:
161            InvalidActionError: When checking is not allowed.
162        """
163        call_amount = self._calculate_call_amount(game_state, player)
164        if call_amount > 0:
165            raise InvalidActionError(f"Cannot check when there is a bet of {call_amount}, must call or fold")
166        
167        return ValidatedAction(
168            original_action=action,
169            actual_action_type=ActionType.CHECK,
170            actual_amount=0,
171            player_seat=player.seat_id
172        )

174    def _validate_call(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
175        """Validate a call action.
176        
177        Args:
178            game_state: The current game state.
179            player: The player attempting to call.
180            action: The call action.
181            
182        Returns:
183            A validated call action, possibly converted to check or all-in.
184        """
185        call_amount = self._calculate_call_amount(game_state, player)
186        
187        if call_amount == 0:
188            # No bet to call, convert to check
189            return ValidatedAction(
190                original_action=action,
191                actual_action_type=ActionType.CHECK,
192                actual_amount=0,
193                player_seat=player.seat_id,
194                is_converted=True,
195                conversion_reason="No bet to call, converted to check"
196            )
197        
198        if player.chips < call_amount:
199            # Insufficient chips, convert to all-in
200            return ValidatedAction(
201                original_action=action,
202                actual_action_type=ActionType.ALL_IN,
203                actual_amount=player.chips,
204                player_seat=player.seat_id,
205                is_converted=True,
206                conversion_reason=f"Insufficient chips to call {call_amount}, converted to all-in {player.chips}"
207            )
208        
209        return ValidatedAction(
210            original_action=action,
211            actual_action_type=ActionType.CALL,
212            actual_amount=call_amount,
213            player_seat=player.seat_id
214        )

216    def _validate_bet(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
217        """Validate a bet action.
218        
219        Args:
220            game_state: The current game state.
221            player: The player attempting to bet.
222            action: The bet action.
223            
224        Returns:
225            A validated bet action, possibly converted to all-in.
226            
227        Raises:
228            InvalidActionError: When betting is not allowed.
229        """
230        # Can only bet when there's no current bet
231        if game_state.current_bet > 0:
232            raise InvalidActionError(
233                f"Cannot bet when there is already a bet of {game_state.current_bet}, "
234                "must call, raise, or fold"
235            )
236        
237        # Check minimum bet amount
238        if action.amount < game_state.big_blind:
239            raise InvalidActionError(
240                f"Bet amount {action.amount} is less than big blind {game_state.big_blind}"
241            )
242        
243        if player.chips < action.amount:
244            # Insufficient chips, convert to all-in
245            return ValidatedAction(
246                original_action=action,
247                actual_action_type=ActionType.ALL_IN,
248                actual_amount=player.chips,
249                player_seat=player.seat_id,
250                is_converted=True,
251                conversion_reason=f"Insufficient chips to bet {action.amount}, converted to all-in {player.chips}"
252            )
253        
254        return ValidatedAction(
255            original_action=action,
256            actual_action_type=ActionType.BET,
257            actual_amount=action.amount,
258            player_seat=player.seat_id
259        )

261    def _validate_raise(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
262        """Validate a raise action.
263        
264        Args:
265            game_state: The current game state.
266            player: The player attempting to raise.
267            action: The raise action.
268            
269        Returns:
270            A validated raise action, possibly converted to all-in.
271            
272        Raises:
273            InvalidActionError: When raising is not allowed.
274            InsufficientChipsError: When the player cannot even call.
275        """
276        if game_state.current_bet == 0:
277            raise InvalidActionError("Cannot raise when there is no bet, must bet instead")
278        
279        # Calculate minimum raise amount
280        last_raise_increment = game_state.last_raise_amount if game_state.last_raise_amount > 0 else game_state.big_blind
281        min_raise_total = game_state.current_bet + last_raise_increment
282        
283        # Calculate player's current bet and potential all-in total
284        player_current_bet = game_state.get_player_current_bet(player.seat_id)
285        all_in_total = player_current_bet + player.chips
286        
287        # Check if this is an all-in situation (player wants to raise to exactly their all-in amount)
288        if action.amount == all_in_total and player.chips > 0:
289            # This is an all-in raise
290            if all_in_total <= game_state.current_bet:
291                # All-in is not even enough to call
292                raise InsufficientChipsError(
293                    f"Insufficient chips to call {game_state.current_bet}, cannot raise"
294                )
295            # All-in raise is valid even if less than minimum raise
296            return ValidatedAction(
297                original_action=action,
298                actual_action_type=ActionType.ALL_IN,
299                actual_amount=all_in_total,
300                player_seat=player.seat_id,
301                is_converted=True,
302                conversion_reason=f"Raise amount {action.amount} equals all-in total, converted to all-in"
303            )
304        
305        # Check if raise amount meets minimum requirement for non-all-in raises
306        if action.amount < min_raise_total:
307            raise InvalidActionError(
308                f"Raise total {action.amount} is less than minimum raise {min_raise_total}. "
309                f"(Current bet: {game_state.current_bet}, minimum increase: {last_raise_increment})"
310            )
311        
312        # Calculate chips needed for this raise
313        chips_needed = action.amount - player_current_bet
314        
315        if player.chips < chips_needed:
316            # Not enough chips for intended raise, convert to all-in
317            if all_in_total < game_state.current_bet and player.chips > 0:
318                raise InsufficientChipsError(
319                    f"Insufficient chips to call {game_state.current_bet}, cannot raise"
320                )
321            
322            return ValidatedAction(
323                original_action=action,
324                actual_action_type=ActionType.ALL_IN,
325                actual_amount=all_in_total,
326                player_seat=player.seat_id,
327                is_converted=True,
328                conversion_reason=f"Insufficient chips to raise to {action.amount}, converted to all-in {all_in_total}"
329            )
330        
331        return ValidatedAction(
332            original_action=action,
333            actual_action_type=ActionType.RAISE,
334            actual_amount=action.amount,
335            player_seat=player.seat_id
336        )

338    def _validate_all_in(self, player: Player, action: Action) -> ValidatedAction:
339        """Validate an all-in action.
340        
341        Args:
342            player: The player attempting to go all-in.
343            action: The all-in action.

345        Returns:

347            
348        Raises:

354        return ValidatedAction(

361    def _calculate_call_amount(self, game_state: GameStateProtocol, player: Player) -> int:
362        """Calculate the amount needed for a player to call.
363        
364        Args:
365            game_state: The current game state.
366            player: The player for whom to calculate the call amount.
367            
368        Returns:
369            The amount needed to call (0 if no call is needed).
370        """
371        player_current_bet = game_state.get_player_current_bet(player.seat_id)
372        return max(0, game_state.current_bet - player_current_bet)

374    def get_available_actions(self, game_state: GameStateProtocol, player: Player) -> List[ActionType]:
375        """Get the list of available actions for a player.
376        
377        Args:
378            game_state: The current game state.
379            player: The player for whom to get available actions.
380            
381        Returns:
382            A list of available action types.
383        """
384        if player.status != SeatStatus.ACTIVE:
385            return []
386        
387        actions = [ActionType.FOLD]  # Can always fold
388        
389        call_amount = self._calculate_call_amount(game_state, player)
390        
391        if call_amount == 0:
392            # No bet to call, can check
393            actions.append(ActionType.CHECK)
394            if player.chips >= game_state.big_blind:
395                actions.append(ActionType.BET)  # Can bet
396        else:
397            # There is a bet, can call
398            actions.append(ActionType.CALL)
399            if player.chips > call_amount:
400                # Can afford more than a call
401                last_raise_increment = game_state.last_raise_amount if game_state.last_raise_amount > 0 else game_state.big_blind
402                min_raise_total = game_state.current_bet + last_raise_increment
403                player_current_bet = game_state.get_player_current_bet(player.seat_id)
404                chips_needed_for_min_raise = min_raise_total - player_current_bet
405                
406                if player.chips >= chips_needed_for_min_raise:
407                    actions.append(ActionType.RAISE)  # Can raise
408        
409        # Can always go all-in if have chips
410        if player.chips > 0:
411            actions.append(ActionType.ALL_IN)
412        
413        return actions

1740    cls = type(self)
1741
1742    if cls._is_protocol:
1743        raise TypeError('Protocols cannot be instantiated')
1744
1745    # Already using a custom `__init__`. No need to calculate correct
1746    # `__init__` to call. This can lead to RecursionError. See bpo-45121.
1747    if cls.__init__ is not _no_init_or_replace_init:
1748        return
1749
1750    # Initially, `__init__` of a protocol subclass is set to `_no_init_or_replace_init`.
1751    # The first instantiation of the subclass will call `_no_init_or_replace_init` which
1752    # searches for a proper new `__init__` in the MRO. The new `__init__`
1753    # replaces the subclass' old `__init__` (ie `_no_init_or_replace_init`). Subsequent
1754    # instantiation of the protocol subclass will thus use the new
1755    # `__init__` and no longer call `_no_init_or_replace_init`.
1756    for base in cls.__mro__:
1757        init = base.__dict__.get('__init__', _no_init_or_replace_init)
1758        if init is not _no_init_or_replace_init:
1759            cls.__init__ = init
1760            break
1761    else:
1762        # should not happen
1763        cls.__init__ = object.__init__
1764

217    def _validate_bet(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
218        """Validate a bet action.
219        
220        Args:
221            game_state: The current game state.
222            player: The player attempting to bet.
223            action: The bet action.
224            
225        Returns:
226            A validated bet action, possibly converted to all-in.
227            
228        Raises:
229            InvalidActionError: When betting is not allowed.
230        """
231        # Can only bet when there's no current bet
232        if game_state.current_bet > 0:
233            raise InvalidActionError(
234                f"Cannot bet when there is already a bet of {game_state.current_bet}, "
235                "must call, raise, or fold"
236            )
237        
238        # Check minimum bet amount
239        if action.amount < game_state.big_blind:
240            raise InvalidActionError(
241                f"Bet amount {action.amount} is less than big blind {game_state.big_blind}"
242            )
243        
244        if player.chips < action.amount:
245            # Insufficient chips, convert to all-in
246            return ValidatedAction(
247                original_action=action,
248                actual_action_type=ActionType.ALL_IN,
249                actual_amount=player.chips,
250                player_seat=player.seat_id,
251                is_converted=True,
252                conversion_reason=f"Insufficient chips to bet {action.amount}, converted to all-in {player.chips}"
253            )
254        
255        return ValidatedAction(
256            original_action=action,
257            actual_action_type=ActionType.BET,
258            actual_amount=action.amount,
259            player_seat=player.seat_id
260        )

262    def _validate_raise(self, game_state: GameStateProtocol, player: Player, action: Action) -> ValidatedAction:
263        """Validate a raise action.
264        
265        Args:
266            game_state: The current game state.
267            player: The player attempting to raise.
268            action: The raise action.
269            
270        Returns:
271            A validated raise action, possibly converted to all-in.
272            
273        Raises:
274            InvalidActionError: When raising is not allowed.
275            InsufficientChipsError: When the player cannot even call.
276        """
277        if game_state.current_bet == 0:
278            raise InvalidActionError("Cannot raise when there is no bet, must bet instead")
279        
280        # Calculate minimum raise amount
281        last_raise_increment = game_state.last_raise_amount if game_state.last_raise_amount > 0 else game_state.big_blind
282        min_raise_total = game_state.current_bet + last_raise_increment
283        
284        # Calculate player's current bet and potential all-in total
285        player_current_bet = game_state.get_player_current_bet(player.seat_id)
286        all_in_total = player_current_bet + player.chips
287        
288        # Check if this is an all-in situation (player wants to raise to exactly their all-in amount)
289        if action.amount == all_in_total and player.chips > 0:
290            # This is an all-in raise
291            if all_in_total <= game_state.current_bet:
292                # All-in is not even enough to call
293                raise InsufficientChipsError(
294                    f"Insufficient chips to call {game_state.current_bet}, cannot raise"
295                )
296            # All-in raise is valid even if less than minimum raise
297            return ValidatedAction(
298                original_action=action,
299                actual_action_type=ActionType.ALL_IN,
300                actual_amount=all_in_total,
301                player_seat=player.seat_id,
302                is_converted=True,
303                conversion_reason=f"Raise amount {action.amount} equals all-in total, converted to all-in"
304            )
305        
306        # Check if raise amount meets minimum requirement for non-all-in raises
307        if action.amount < min_raise_total:
308            raise InvalidActionError(
309                f"Raise total {action.amount} is less than minimum raise {min_raise_total}. "
310                f"(Current bet: {game_state.current_bet}, minimum increase: {last_raise_increment})"
311            )
312        
313        # Calculate chips needed for this raise
314        chips_needed = action.amount - player_current_bet
315        
316        if player.chips < chips_needed:
317            # Not enough chips for intended raise, convert to all-in
318            if all_in_total < game_state.current_bet and player.chips > 0:
319                raise InsufficientChipsError(
320                    f"Insufficient chips to call {game_state.current_bet}, cannot raise"
321                )
322            
323            return ValidatedAction(
324                original_action=action,
325                actual_action_type=ActionType.ALL_IN,
326                actual_amount=all_in_total,
327                player_seat=player.seat_id,
328                is_converted=True,
329                conversion_reason=f"Insufficient chips to raise to {action.amount}, converted to all-in {all_in_total}"
330            )
331        
332        return ValidatedAction(
333            original_action=action,
334            actual_action_type=ActionType.RAISE,
335            actual_amount=action.amount,
336            player_seat=player.seat_id
337        )

362    def _calculate_call_amount(self, game_state: GameStateProtocol, player: Player) -> int:
363        """Calculate the amount needed for a player to call.
364        
365        Args:
366            game_state: The current game state.
367            player: The player for whom to calculate the call amount.

375    def get_available_actions(self, game_state: GameStateProtocol, player: Player) -> List[ActionType]:
376        """Get the list of available actions for a player.
377        
378        Args:
379            game_state: The current game state.
380            player: The player for whom to get available actions.
381            
382        Returns:
383            A list of available action types.
384        """
385        if player.status != SeatStatus.ACTIVE:
386            return []
387        
388        actions = [ActionType.FOLD]  # Can always fold
389        
390        call_amount = self._calculate_call_amount(game_state, player)
391        
392        if call_amount == 0:

394            actions.append(ActionType.CHECK)
395            if player.chips >= game_state.big_blind:
396                actions.append(ActionType.BET)  # Can bet
397        else:
398            # There is a bet, can call
399            actions.append(ActionType.CALL)
400            if player.chips > call_amount:
401                # Can afford more than a call
402                last_raise_increment = game_state.last_raise_amount if game_state.last_raise_amount > 0 else game_state.big_blind
403                min_raise_total = game_state.current_bet + last_raise_increment
404                player_current_bet = game_state.get_player_current_bet(player.seat_id)
405                chips_needed_for_min_raise = min_raise_total - player_current_bet
406                
407                if player.chips >= chips_needed_for_min_raise:
408                    actions.append(ActionType.RAISE)  # Can raise
409        
410        # Can always go all-in if have chips
411        if player.chips > 0:
412            actions.append(ActionType.ALL_IN)
413        
414        return actions
