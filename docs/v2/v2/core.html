 27    def __str__(self) -> str:
 28        """返回牌的字符串表示。

 30        Returns:
 31            格式为"点数花色"的字符串，如"AH"表示红桃A

 33        rank_display = {

 53    @classmethod
 54    def from_str(cls, card_str: str) -> 'Card':
 55        """从字符串创建Card对象。
 56        
 57        Args:
 58            card_str: 牌的字符串表示，如"As"表示黑桃A，"Kh"表示红桃K
 59            
 60        Returns:
 61            对应的Card对象
 62            
 63        Raises:
 64            ValueError: 如果字符串格式无效
 65        """
 66        if len(card_str) < 2:
 67            raise ValueError(f"卡牌字符串格式错误: {card_str}")
 68        
 69        # 处理10的特殊情况
 70        if card_str.startswith("10"):
 71            rank_str, suit_str = "10", card_str[2:]
 72        else:
 73            rank_str, suit_str = card_str[0], card_str[1:]
 74        
 75        # 解析点数
 76        rank_map = {
 77            "2": Rank.TWO, "3": Rank.THREE, "4": Rank.FOUR, "5": Rank.FIVE,
 78            "6": Rank.SIX, "7": Rank.SEVEN, "8": Rank.EIGHT, "9": Rank.NINE,
 79            "10": Rank.TEN, "T": Rank.TEN, "J": Rank.JACK, "Q": Rank.QUEEN,
 80            "K": Rank.KING, "A": Rank.ACE
 81        }
 82        
 83        # 解析花色
 84        suit_map = {
 85            "h": Suit.HEARTS, "H": Suit.HEARTS,
 86            "d": Suit.DIAMONDS, "D": Suit.DIAMONDS,
 87            "c": Suit.CLUBS, "C": Suit.CLUBS,
 88            "s": Suit.SPADES, "S": Suit.SPADES
 89        }
 90        
 91        if rank_str not in rank_map:
 92            raise ValueError(f"无效的点数: {rank_str}")
 93        if suit_str not in suit_map:
 94            raise ValueError(f"无效的花色: {suit_str}")
 95        
 96        return cls(suit_map[suit_str], rank_map[rank_str])

 98    def __lt__(self, other: 'Card') -> bool:
 99        """比较两张牌的大小（仅比较点数）。

133class Deck:
134    """
135    表示一副扑克牌。

140    def __init__(self, rng: Optional[random.Random] = None):
141        """
142        初始化一副完整的52张牌。
143        
144        Args:
145            rng: 随机数生成器，用于洗牌。如果为None则使用默认随机数生成器
146        """
147        self._rng = rng or random.Random()
148        self._cards: List[Card] = []
149        self._reset_deck()
150    
151    def _reset_deck(self) -> None:
152        """重置牌堆为完整的52张牌。"""
153        self._cards = [
154            Card(suit, rank) 
155            for suit in Suit 
156            for rank in Rank
157        ]
158    
159    def shuffle(self) -> None:
160        """洗牌。"""
161        self._rng.shuffle(self._cards)
162    
163    def deal_card(self) -> Card:
164        """
165        发一张牌。
166        
167        Returns:
168            发出的牌
169            
170        Raises:
171            IndexError: 如果牌堆为空
172        """
173        if not self._cards:
174            raise IndexError("Cannot deal from empty deck")
175        return self._cards.pop()
176    
177    def deal_cards(self, count: int) -> List[Card]:
178        """
179        发多张牌。
180        
181        Args:
182            count: 要发的牌数
183            
184        Returns:
185            发出的牌列表
186            
187        Raises:
188            ValueError: 如果要发的牌数超过剩余牌数
189        """
190        if count < 0:
191            raise ValueError("Count must be non-negative")
192        if count > len(self._cards):
193            raise ValueError(f"Cannot deal {count} cards, only {len(self._cards)} remaining")
194        
195        dealt_cards = []
196        for _ in range(count):
197            dealt_cards.append(self.deal_card())
198        return dealt_cards
199    
200    def cards_remaining(self) -> int:
201        """
202        返回剩余牌数。
203        
204        Returns:
205            剩余的牌数
206        """
207        return len(self._cards)
208    
209    def is_empty(self) -> bool:
210        """
211        检查牌堆是否为空。
212        
213        Returns:
214            如果牌堆为空则返回True
215        """
216        return len(self._cards) == 0
217    
218    def reset(self) -> None:
219        """重置牌堆为完整的52张牌（未洗牌）。"""
220        self._reset_deck()
221    
222    def peek_top(self) -> Optional[Card]:

228        """
229        return self._cards[-1] if self._cards else None

231    def __len__(self) -> int:
232        """返回牌堆中的牌数。"""

182            raise ValueError(f"必须是5张牌，实际: {len(cards)}")
183        
184        # 按点数降序排序
185        sorted_cards = sorted(cards, key=lambda c: c.rank.value, reverse=True)
186        ranks = [card.rank.value for card in sorted_cards]
187        suits = [card.suit for card in sorted_cards]
188

190        rank_counts = CollectionsCounter(ranks)
191        count_groups = CollectionsCounter(rank_counts.values())
192

204            return HandResult(HandRank.STRAIGHT_FLUSH, straight_high)
205

208            kicker = self._get_rank_with_count(rank_counts, 1)
209            return HandResult(HandRank.FOUR_OF_A_KIND, four_rank, kickers=(kicker,))
210

212            three_rank = self._get_rank_with_count(rank_counts, 3)

218

221        
222        if count_groups[3] == 1:  # 三条

224            kickers = tuple(sorted([r for r in ranks if r != three_rank], reverse=True))
225            return HandResult(HandRank.THREE_OF_A_KIND, three_rank, kickers=kickers)
226        
227        if count_groups[2] == 2:  # 两对
228            pairs = sorted([r for r, c in rank_counts.items() if c == 2], reverse=True)
229            kicker = self._get_rank_with_count(rank_counts, 1)
230            return HandResult(HandRank.TWO_PAIR, pairs[0], pairs[1], kickers=(kicker,))
231        
232        if count_groups[2] == 1:  # 一对
233            pair_rank = self._get_rank_with_count(rank_counts, 2)
234            kickers = tuple(sorted([r for r in ranks if r != pair_rank], reverse=True))
235            return HandResult(HandRank.ONE_PAIR, pair_rank, kickers=kickers)
236        
237        # 高牌
238        return HandResult(HandRank.HIGH_CARD, ranks[0], kickers=tuple(ranks[1:]))
