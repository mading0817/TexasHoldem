  77    def start_new_hand(self) -> bool:
  78        """开始新的一手牌.
  79        
  80        初始化新手牌的所有必要状态，包括：
  81        - 重置玩家状态
  82        - 发放底牌
  83        - 收取盲注
  84        - 设置第一个行动玩家
  85        
  86        Returns:
  87            是否成功开始新手牌
  88            
  89        Raises:
  90            RuntimeError: 如果当前已有手牌在进行中
  91        """
  92        if self._hand_in_progress:
  93            raise RuntimeError("当前已有手牌在进行中，无法开始新手牌")
  94            
  95        # 预先重置游戏状态，以便正确检查活跃玩家
  96        self._advance_dealer_position()
  97        
  98        # 重置所有玩家状态为新手牌
  99        for player in self._game_state.players:
 100            player.reset_for_new_hand()
 101            player.is_dealer = False
 102            player.is_small_blind = False
 103            player.is_big_blind = False
 104            player.total_bet_this_hand = 0
 105        
 106        # 设置庄家标记
 107        if self._game_state.players:
 108            dealer_player = self._game_state.get_player_by_seat(self._game_state.dealer_position)
 109            if dealer_player:
 110                dealer_player.is_dealer = True
 111        
 112        # 检查是否有足够的活跃玩家（重置后检查）
 113        active_players = [p for p in self._game_state.players if p.status == SeatStatus.ACTIVE and p.chips > 0]
 114        if len(active_players) < 2:
 115            self._logger.info("活跃玩家不足2人，无法开始新手牌")
 116            return False
 117            
 118        # 发射手牌开始事件
 119        self._event_bus.emit_simple(
 120            EventType.HAND_STARTED,
 121            active_players=len(active_players),
 122            dealer_position=self._game_state.dealer_position
 123        )
 124        
 125        # 完成其余的重置工作
 126        self._complete_hand_reset()
 127        self._hand_in_progress = True
 128        
 129        self._logger.info(f"开始新手牌，活跃玩家数: {len(active_players)}")
 130        return True

 133    def execute_action(self, action: Action) -> bool:
 134        """执行玩家行动.
 135        
 136        验证并执行玩家的行动，更新游戏状态。
 137        
 138        Args:
 139            action: 玩家要执行的行动
 140            
 141        Returns:
 142            是否成功执行行动
 143            
 144        Raises:
 145            ValueError: 如果行动无效
 146        """
 147        if not self._hand_in_progress:
 148            raise RuntimeError("当前没有手牌在进行中")
 149            
 150        # 验证行动
 151        player = self._game_state.get_player_by_seat(action.player_id)
 152        if player is None:
 153            raise ValueError(f"找不到玩家 {action.player_id}")
 154            
 155        validated_action = self._validator.validate(self._game_state, player, action)
 156        
 157        if not validated_action.validation_result.is_valid:
 158            error_msg = validated_action.validation_result.error_message or "行动无效"
 159            self._logger.warning(f"行动验证失败: {error_msg}")
 160            raise ValueError(error_msg)
 161            
 162        # 执行验证后的行动
 163        final_action = validated_action.final_action
 164        self._apply_action(final_action)
 165        
 166        self._logger.info(f"玩家{final_action.player_id}执行行动: {final_action.action_type.value}")
 167        
 168        # 检查是否需要进入下一阶段或结束手牌
 169        self._check_phase_transition()
 170        
 171        return True

 186        """

 191        active_players = [
 192            p for p in self._game_state.players 
 193            if p.status in [SeatStatus.ACTIVE, SeatStatus.ALL_IN]
 194        ]
 195        
 196        if len(active_players) <= 1:
 197            return True

 199        # 只有在SHOWDOWN阶段才认为手牌结束
 200        if self._game_state.phase == Phase.SHOWDOWN:
 201            return True

 203        return False

 240    def end_hand(self) -> Optional[HandResult]:
 241        """结束当前手牌并计算结果.
 242        
 243        Returns:
 244            手牌结果，如果没有手牌在进行中则返回None
 245        """
 246        if not self._hand_in_progress:
 247            return None
 248            
 249        # 收集所有下注到底池
 250        self._game_state.collect_bets_to_pot()
 251        
 252        # 获取所有未弃牌的玩家
 253        active_players = [
 254            p for p in self._game_state.players 
 255            if p.status in [SeatStatus.ACTIVE, SeatStatus.ALL_IN]
 256        ]
 257        
 258        if len(active_players) == 0:
 259            # 没有活跃玩家，这种情况不应该发生
 260            self._hand_in_progress = False
 261            self._logger.warning("手牌结束时没有活跃玩家")
 262            # 移除庄家位置移动 - 应该在下一手牌开始时移动
 263            return HandResult(
 264                winner_ids=[],
 265                pot_amount=self._game_state.pot,
 266                winning_hand_description="无活跃玩家",
 267                side_pots=[]
 268            )
 269        
 270        if len(active_players) == 1:
 271            # 只有一个玩家，直接获胜
 272            winner = active_players[0]
 273            winner.chips += self._game_state.pot
 274            
 275            result = HandResult(
 276                winner_ids=[winner.seat_id],
 277                pot_amount=self._game_state.pot,
 278                winning_hand_description=f"{winner.name} 获胜（其他玩家弃牌）",
 279                side_pots=[]
 280            )
 281            
 282            self._game_state.pot = 0  # 清空底池
 283            self._hand_in_progress = False
 284            self._logger.info(f"{winner.name} 获胜，获得底池 {result.pot_amount}")
 285            
 286            # 移除庄家位置移动 - 应该在下一手牌开始时移动
 287            
 288            # 发射手牌结束事件
 289            self._event_bus.emit_simple(
 290                EventType.HAND_ENDED,
 291                winner_ids=result.winner_ids,
 292                pot_amount=result.pot_amount,
 293                winning_hand_description=result.winning_hand_description,
 294                side_pots=result.side_pots
 295            )
 296            
 297            return result
 298        
 299        # 多个玩家，需要比较牌型
 300        from v2.core.evaluator import SimpleEvaluator
 301        evaluator = SimpleEvaluator()
 302        
 303        # 评估每个玩家的牌型
 304        player_hands = []
 305        for player in active_players:
 306            try:
 307                hand_result = evaluator.evaluate_hand(
 308                    player.hole_cards, 
 309                    self._game_state.community_cards
 310                )
 311                player_hands.append((player, hand_result))
 312                self._logger.info(f"{player.name} 的牌型: {hand_result}")
 313            except Exception as e:
 314                self._logger.error(f"评估玩家 {player.name} 牌型失败: {e}")
 315                # 如果评估失败，给予最低牌型
 316                from v2.core.evaluator import HandResult as EvalHandResult
 317                from v2.core.enums import HandRank
 318                hand_result = EvalHandResult(HandRank.HIGH_CARD, 2)
 319                player_hands.append((player, hand_result))
 320        
 321        # 找出最佳牌型
 322        best_hand = max(player_hands, key=lambda x: (x[1].rank.value, x[1].primary_value, x[1].secondary_value, x[1].kickers))
 323        best_hand_result = best_hand[1]
 324        
 325        # 找出所有拥有最佳牌型的玩家（可能平局）
 326        winners = []
 327        for player, hand_result in player_hands:
 328            if hand_result.compare_to(best_hand_result) == 0:
 329                winners.append(player)
 330        
 331        # 分配底池
 332        pot_per_winner = self._game_state.pot // len(winners)
 333        remainder = self._game_state.pot % len(winners)
 334        
 335        winner_ids = []
 336        for i, winner in enumerate(winners):
 337            share = pot_per_winner
 338            if i < remainder:  # 余数分配给前几个获胜者
 339                share += 1
 340            winner.chips += share
 341            winner_ids.append(winner.seat_id)
 342            self._logger.info(f"{winner.name} 获得 {share} 筹码")
 343        
 344        # 构建获胜描述
 345        if len(winners) == 1:
 346            winning_description = f"{winners[0].name} 获胜 - {best_hand_result}"
 347        else:
 348            winner_names = [w.name for w in winners]
 349            winning_description = f"平局: {', '.join(winner_names)} - {best_hand_result}"
 350        
 351        result = HandResult(
 352            winner_ids=winner_ids,
 353            pot_amount=self._game_state.pot,
 354            winning_hand_description=winning_description,
 355            side_pots=[]  # TODO: 实现边池逻辑
 356        )
 357        
 358        # 清空底池
 359        self._game_state.pot = 0
 360        self._hand_in_progress = False
 361        self._logger.info(f"手牌结束: {winning_description}")
 362        
 363        # 移除庄家位置移动 - 应该在下一手牌开始时移动
 364        
 365        # 发射手牌结束事件
 366        self._event_bus.emit_simple(
 367            EventType.HAND_ENDED,
 368            winner_ids=result.winner_ids,
 369            pot_amount=result.pot_amount,
 370            winning_hand_description=result.winning_hand_description,
 371            side_pots=result.side_pots
 372        )
 373        
 374        return result

 446    def _post_blinds(self) -> None:
 447        """收取盲注."""
 448        players = self._game_state.players
 449        if len(players) < 2:
 450            return
 451            
 452        # 找到小盲和大盲位置
 453        dealer_pos = self._game_state.dealer_position
 454        small_blind_pos = (dealer_pos + 1) % len(players)
 455        big_blind_pos = (dealer_pos + 2) % len(players)
 456        
 457        # 获取当前阶段标识
 458        phase_prefix = f"[{self._game_state.phase.value}] "
 459        
 460        # 收取小盲注
 461        small_blind_player = players[small_blind_pos]
 462        if small_blind_player.chips >= self._game_state.small_blind:
 463            small_blind_player.bet(self._game_state.small_blind)
 464            small_blind_player.is_small_blind = True
 465            self._game_state.add_event(f"{phase_prefix}{small_blind_player.name} 下小盲注 {self._game_state.small_blind}")
 466        
 467        # 收取大盲注
 468        big_blind_player = players[big_blind_pos]
 469        if big_blind_player.chips >= self._game_state.big_blind:
 470            big_blind_player.bet(self._game_state.big_blind)
 471            big_blind_player.is_big_blind = True
 472            self._game_state.current_bet = self._game_state.big_blind
 473            self._game_state.add_event(f"{phase_prefix}{big_blind_player.name} 下大盲注 {self._game_state.big_blind}")
 474            
 475        # 发射盲注事件
 476        self._event_bus.emit_simple(
 477            EventType.BLINDS_POSTED,
 478            small_blind_player_id=small_blind_pos,
 479            small_blind_amount=self._game_state.small_blind,
 480            big_blind_player_id=big_blind_pos,
 481            big_blind_amount=self._game_state.big_blind
 482        )
 483        
 484    def _set_first_player(self) -> None:
 485        """设置第一个行动玩家."""
 486        players = self._game_state.players
 487        if len(players) < 3:
 488            # 如果只有两个玩家，小盲先行动
 489            self._game_state.current_player = (self._game_state.dealer_position + 1) % len(players)
 490        else:
 491            # 如果有三个或更多玩家，大盲后面的玩家先行动
 492            self._game_state.current_player = (self._game_state.dealer_position + 3) % len(players)

 494    def _apply_action(self, action: Action) -> None:
 495        """应用玩家行动到游戏状态."""
 496        player = self._game_state.get_player_by_seat(action.player_id)
 497        if player is None:
 498            raise ValueError(f"找不到玩家 {action.player_id}")
 499        
 500        # 获取当前阶段标识
 501        phase_prefix = f"[{self._game_state.phase.value}] "
 502            
 503        if action.action_type == ActionType.FOLD:
 504            player.status = SeatStatus.FOLDED
 505            self._game_state.add_event(f"{phase_prefix}{player.name} 弃牌")
 506            # 发射玩家弃牌事件
 507            self._event_bus.emit_simple(
 508                EventType.PLAYER_FOLDED,
 509                player_id=action.player_id,
 510                player_name=player.name
 511            )
 512            
 513        elif action.action_type == ActionType.CHECK:
 514            self._game_state.add_event(f"{phase_prefix}{player.name} 过牌")
 515            # 发射玩家行动事件
 516            self._event_bus.emit_simple(
 517                EventType.PLAYER_ACTION,
 518                player_id=action.player_id,
 519                player_name=player.name,
 520                action_type="check",
 521                amount=0
 522            )
 523            
 524        elif action.action_type == ActionType.CALL:
 525            call_amount = self._game_state.current_bet - player.current_bet
 526            if call_amount > 0:
 527                player.bet(call_amount)
 528                # 修复：显示"跟注到X"而不是"跟注X"
 529                self._game_state.add_event(f"{phase_prefix}{player.name} 跟注到 {self._game_state.current_bet}")
 530                # 发射玩家行动事件
 531                self._event_bus.emit_simple(
 532                    EventType.PLAYER_ACTION,
 533                    player_id=action.player_id,
 534                    player_name=player.name,
 535                    action_type="call",
 536                    amount=call_amount
 537                )
 538            else:
 539                self._game_state.add_event(f"{phase_prefix}{player.name} 过牌")
 540                # 发射玩家行动事件
 541                self._event_bus.emit_simple(
 542                    EventType.PLAYER_ACTION,
 543                    player_id=action.player_id,
 544                    player_name=player.name,
 545                    action_type="check",
 546                    amount=0
 547                )
 548                
 549        elif action.action_type == ActionType.BET:
 550            total_bet = action.amount
 551            bet_amount = total_bet - player.current_bet
 552            player.bet(bet_amount)
 553            self._game_state.current_bet = total_bet
 554            self._game_state.last_raiser = action.player_id
 555            # BET时，加注增量就是下注金额（相对于0）
 556            self._game_state.last_raise_amount = total_bet
 557            self._game_state.add_event(f"{phase_prefix}{player.name} 下注 {total_bet}")
 558            # 发射下注事件
 559            self._event_bus.emit_simple(
 560                EventType.BET_PLACED,
 561                player_id=action.player_id,
 562                player_name=player.name,
 563                action_type="bet",
 564                amount=bet_amount,
 565                total_bet=total_bet,
 566                new_current_bet=self._game_state.current_bet
 567            )
 568            
 569        elif action.action_type == ActionType.RAISE:
 570            total_bet = action.amount
 571            bet_amount = total_bet - player.current_bet
 572            player.bet(bet_amount)
 573            # 计算实际加注增量（新的总下注 - 之前的总下注）
 574            previous_bet = self._game_state.current_bet
 575            raise_increment = total_bet - previous_bet
 576            self._game_state.current_bet = total_bet
 577            self._game_state.last_raiser = action.player_id
 578            self._game_state.last_raise_amount = raise_increment
 579            self._game_state.add_event(f"{phase_prefix}{player.name} 加注到 {total_bet}")
 580            # 发射下注事件
 581            self._event_bus.emit_simple(
 582                EventType.BET_PLACED,
 583                player_id=action.player_id,
 584                player_name=player.name,
 585                action_type="raise",
 586                amount=bet_amount,
 587                total_bet=total_bet,
 588                new_current_bet=self._game_state.current_bet
 589            )
 590            
 591        elif action.action_type == ActionType.ALL_IN:
 592            all_in_amount = player.chips
 593            player.bet(all_in_amount)
 594            player.status = SeatStatus.ALL_IN
 595            if player.current_bet > self._game_state.current_bet:
 596                self._game_state.current_bet = player.current_bet
 597                self._game_state.last_raiser = action.player_id
 598            self._game_state.add_event(f"{phase_prefix}{player.name} 全押 {all_in_amount}")
 599            # 发射全押事件
 600            self._event_bus.emit_simple(
 601                EventType.PLAYER_ALL_IN,
 602                player_id=action.player_id,
 603                player_name=player.name,
 604                amount=all_in_amount,
 605                new_current_bet=self._game_state.current_bet
 606            )
 607            
 608        # 移动到下一个玩家
 609        self._advance_to_next_player()
 610        
 611        # 增加行动计数
 612        self._game_state.increment_action_count()
 613        
 614        # 发射底池更新事件
 615        self._event_bus.emit_simple(
 616            EventType.POT_UPDATED,
 617            pot_amount=self._game_state.pot,
 618            current_bet=self._game_state.current_bet
 619        )

 646    def _all_actions_complete(self) -> bool:
 647        """检查当前阶段是否所有行动都已完成.
 648        
 649        德州扑克规则：
 650        1. 每个活跃玩家在每个下注轮都必须有机会行动
 651        2. 只有当所有活跃玩家都匹配了当前下注且都已行动过时，下注轮才结束
 652        3. 如果只剩一个活跃玩家，手牌立即结束（不是阶段结束）
 653        4. 在PRE_FLOP阶段，大盲注玩家有最后行动权
 654        """
 655        active_players = self._game_state.get_active_players()
 656        
 657        # 调试信息
 658        self._logger.debug(f"检查下注轮完成: 活跃玩家数={len(active_players)}, 当前下注={self._game_state.current_bet}, 行动数={self._game_state.actions_this_round}, 当前玩家={self._game_state.current_player}, 加注者={self._game_state.last_raiser}")
 659        
 660        # 如果没有活跃玩家，行动完成
 661        if not active_players:
 662            self._logger.debug("没有活跃玩家，下注轮完成")
 663            return True
 664        
 665        # 如果只有一个活跃玩家，手牌应该结束（不是阶段结束）
 666        # 这种情况应该在is_hand_over()中处理，而不是在这里
 667        if len(active_players) == 1:
 668            # 检查是否还有ALL_IN玩家
 669            all_in_players = [p for p in self._game_state.players if p.status == SeatStatus.ALL_IN]
 670            if not all_in_players:
 671                # 只有一个活跃玩家且没有ALL_IN玩家，手牌结束
 672                self._logger.debug("只有一个活跃玩家且无ALL_IN玩家，手牌结束")
 673                return True
 674        
 675        # 检查是否所有活跃玩家都已匹配当前下注
 676        for player in active_players:
 677            if player.current_bet < self._game_state.current_bet:
 678                self._logger.debug(f"玩家{player.seat_id}下注{player.current_bet}未匹配当前下注{self._game_state.current_bet}")
 679                return False
 680        
 681        # 所有玩家都匹配了当前下注，检查行动是否完成
 682        self._logger.debug("所有活跃玩家都匹配了当前下注")
 683        
 684        # 计算需要的最小行动数
 685        min_actions_needed = len(active_players)
 686        
 687        # 在PRE_FLOP阶段，需要特殊处理大盲注的最后行动权
 688        if self._game_state.phase == Phase.PRE_FLOP:
 689            # 如果行动数不足，下注轮未完成
 690            if self._game_state.actions_this_round < min_actions_needed:
 691                self._logger.debug(f"PRE_FLOP阶段行动数不足: {self._game_state.actions_this_round} < {min_actions_needed}")
 692                return False
 693            
 694            # 如果没有加注者，大盲注玩家有最后行动权
 695            if self._game_state.last_raiser is None:
 696                # 找到大盲注玩家
 697                big_blind_player = None
 698                for player in self._game_state.players:
 699                    if hasattr(player, 'is_big_blind') and player.is_big_blind:
 700                        big_blind_player = player
 701                        break
 702                
 703                # 如果大盲注玩家还是当前玩家且还没有行动过，需要等待
 704                if (big_blind_player and 
 705                    big_blind_player.status == SeatStatus.ACTIVE and 
 706                    self._game_state.current_player == big_blind_player.seat_id and
 707                    not hasattr(big_blind_player, 'has_acted_this_round')):
 708                    self._logger.debug("PRE_FLOP阶段大盲注玩家还未行动")
 709                    return False
 710        else:
 711            # 其他阶段：确保每个活跃玩家都至少行动过一次
 712            if self._game_state.actions_this_round < min_actions_needed:
 713                self._logger.debug(f"非PRE_FLOP阶段行动数不足: {self._game_state.actions_this_round} < {min_actions_needed}")
 714                return False
 715        
 716        # 关键修复：如果有加注者，检查是否轮回到加注者后的第一个活跃玩家
 717        if self._game_state.last_raiser is not None:
 718            players = self._game_state.players
 719            raiser_pos = self._game_state.last_raiser
 720            
 721            # 找到加注者后的第一个活跃玩家
 722            next_active_after_raiser = None
 723            for i in range(1, len(players) + 1):
 724                next_pos = (raiser_pos + i) % len(players)
 725                if players[next_pos].status == SeatStatus.ACTIVE:
 726                    next_active_after_raiser = next_pos
 727                    break
 728            
 729            self._logger.debug(f"有加注者{raiser_pos}, 加注者后第一个活跃玩家{next_active_after_raiser}, 当前玩家{self._game_state.current_player}")
 730            
 731            # 如果当前玩家就是加注者后的第一个活跃玩家，且所有人都匹配了下注，下注轮完成
 732            if (next_active_after_raiser is not None and 
 733                self._game_state.current_player == next_active_after_raiser):
 734                self._logger.debug("轮回到加注者后第一个活跃玩家，下注轮完成")
 735                return True
 736            else:
 737                self._logger.debug("未轮回到加注者后第一个活跃玩家，下注轮未完成")
 738                return False
 739        else:
 740            # 没有加注者的情况：如果所有人都匹配下注且行动数足够，下注轮完成
 741            result = self._game_state.actions_this_round >= min_actions_needed
 742            self._logger.debug(f"无加注者，基于行动数判断: {result}")
 743            return result

 794            )
 795            # 发射发牌事件
 796            self._event_bus.emit_simple(
 797                EventType.CARDS_DEALT,
 798                phase=self._game_state.phase.value,
 799                cards_count=1,
 800                community_cards_count=len(self._game_state.community_cards)
 801            )

 805            self._event_bus.emit_simple(
 806                EventType.PHASE_CHANGED,
 807                from_phase=old_phase.value,
 808                to_phase=self._game_state.phase.value
 809            )
 810            
 811        # 重置下注轮并设置第一个行动玩家
 812        if self._game_state.phase != Phase.SHOWDOWN:
 813            self._game_state.reset_betting_round()
 814            self._set_first_player_for_new_round()

这个类是游戏的主要控制器，负责：
