 93    def start_new_hand(self) -> bool:
 94        """开始新的一手牌.
 95        
 96        初始化新手牌的所有必要状态，包括：
 97        - 重置玩家状态
 98        - 发放底牌
 99        - 收取盲注
100        - 设置第一个行动玩家
101        
102        Returns:
103            是否成功开始新手牌
104            
105        Raises:
106            RuntimeError: 如果当前已有手牌在进行中
107        """
108        if self._hand_in_progress:
109            raise RuntimeError("当前已有手牌在进行中，无法开始新手牌")
110            
111        # 检查是否有足够的活跃玩家
112        active_players = [p for p in self._game_state.players if p.status == SeatStatus.ACTIVE and p.chips > 0]
113        if len(active_players) < 2:
114            self._logger.info("活跃玩家不足2人，无法开始新手牌")
115            return False
116            
117        # 重置游戏状态为新手牌
118        self._reset_for_new_hand()
119        self._hand_in_progress = True
120        
121        self._logger.info(f"开始新手牌，活跃玩家数: {len(active_players)}")
122        return True

124    def execute_action(self, action: Action) -> bool:
125        """执行玩家行动.
126        
127        验证并执行玩家的行动，更新游戏状态。
128        
129        Args:
130            action: 玩家要执行的行动
131            
132        Returns:
133            是否成功执行行动
134            
135        Raises:
136            ValueError: 如果行动无效
137        """
138        if not self._hand_in_progress:
139            raise RuntimeError("当前没有手牌在进行中")
140            
141        # 验证行动
142        validated_action = self._validator.validate_action(self._game_state, action)
143        
144        if not validated_action.validation_result.is_valid:
145            error_msg = validated_action.validation_result.error_message or "行动无效"
146            self._logger.warning(f"行动验证失败: {error_msg}")
147            raise ValueError(error_msg)
148            
149        # 执行验证后的行动
150        final_action = validated_action.final_action
151        self._apply_action(final_action)
152        
153        self._logger.info(f"玩家{final_action.player_id}执行行动: {final_action.action_type.value}")
154        
155        # 检查是否需要进入下一阶段或结束手牌
156        self._check_phase_transition()
157        
158        return True

272    def _post_blinds(self) -> None:
273        """收取盲注."""
274        players = self._game_state.players
275        if len(players) < 2:
276            return
277            
278        # 找到小盲和大盲位置
279        dealer_pos = self._game_state.dealer_position
280        small_blind_pos = (dealer_pos + 1) % len(players)
281        big_blind_pos = (dealer_pos + 2) % len(players)
282        
283        # 收取小盲注
284        small_blind_player = players[small_blind_pos]
285        if small_blind_player.chips >= self._game_state.small_blind:
286            small_blind_player.bet(self._game_state.small_blind)
287            small_blind_player.is_small_blind = True
288            self._game_state.add_event(f"{small_blind_player.name} 下小盲注 {self._game_state.small_blind}")
289        
290        # 收取大盲注
291        big_blind_player = players[big_blind_pos]
292        if big_blind_player.chips >= self._game_state.big_blind:
293            big_blind_player.bet(self._game_state.big_blind)
294            big_blind_player.is_big_blind = True
295            self._game_state.current_bet = self._game_state.big_blind
296            self._game_state.add_event(f"{big_blind_player.name} 下大盲注 {self._game_state.big_blind}")
297            
298    def _set_first_player(self) -> None:
299        """设置第一个行动玩家."""
300        players = self._game_state.players
301        if len(players) < 3:
302            # 如果只有两个玩家，小盲先行动
303            self._game_state.current_player = (self._game_state.dealer_position + 1) % len(players)
304        else:
305            # 如果有三个或更多玩家，大盲后面的玩家先行动
306            self._game_state.current_player = (self._game_state.dealer_position + 3) % len(players)
307            
308    def _apply_action(self, action: Action) -> None:
309        """应用玩家行动到游戏状态."""
310        player = self._game_state.get_player_by_seat(action.player_id)
311        if player is None:
312            raise ValueError(f"找不到玩家 {action.player_id}")
313            
314        if action.action_type == ActionType.FOLD:
315            player.status = SeatStatus.FOLDED
316            self._game_state.add_event(f"{player.name} 弃牌")
317            
318        elif action.action_type == ActionType.CHECK:
319            self._game_state.add_event(f"{player.name} 过牌")
320            
321        elif action.action_type == ActionType.CALL:
322            call_amount = self._game_state.current_bet - player.current_bet
323            if call_amount > 0:
324                player.bet(call_amount)
325                self._game_state.add_event(f"{player.name} 跟注 {call_amount}")
326            else:
327                self._game_state.add_event(f"{player.name} 过牌")
328                
329        elif action.action_type == ActionType.BET:
330            player.bet(action.amount)
331            self._game_state.current_bet = action.amount
332            self._game_state.last_raiser = action.player_id
333            self._game_state.add_event(f"{player.name} 下注 {action.amount}")
334            
335        elif action.action_type == ActionType.RAISE:
336            total_bet = self._game_state.current_bet + action.amount
337            bet_amount = total_bet - player.current_bet
338            player.bet(bet_amount)
339            self._game_state.current_bet = total_bet
340            self._game_state.last_raiser = action.player_id
341            self._game_state.last_raise_amount = action.amount
342            self._game_state.add_event(f"{player.name} 加注到 {total_bet}")
343            
344        elif action.action_type == ActionType.ALL_IN:
345            all_in_amount = player.chips
346            player.bet(all_in_amount)
347            player.status = SeatStatus.ALL_IN
348            if player.current_bet > self._game_state.current_bet:
349                self._game_state.current_bet = player.current_bet
350                self._game_state.last_raiser = action.player_id
351            self._game_state.add_event(f"{player.name} 全押 {all_in_amount}")
352            
353        # 移动到下一个玩家
354        self._advance_to_next_player()
355        
356    def _advance_to_next_player(self) -> None:
357        """移动到下一个需要行动的玩家."""
358        if self._game_state.current_player is None:
359            return
360            
361        players = self._game_state.players
362        start_pos = self._game_state.current_player
363        
364        # 寻找下一个可以行动的玩家
365        for i in range(1, len(players) + 1):
366            next_pos = (start_pos + i) % len(players)
367            next_player = players[next_pos]
368            
369            if next_player.status == SeatStatus.ACTIVE:
370                self._game_state.current_player = next_pos
371                return
372                
373        # 如果没有找到活跃玩家，设置为None
374        self._game_state.current_player = None
375        
376    def _check_phase_transition(self) -> None:
377        """检查并处理游戏阶段转换."""
378        if self._all_actions_complete():
379            self._advance_to_next_phase()
380            
381    def _all_actions_complete(self) -> bool:
382        """检查当前阶段是否所有行动都已完成."""
383        active_players = self._game_state.get_active_players()
384        
385        # 如果没有活跃玩家，行动完成
386        if not active_players:
387            return True
388            
389        # 检查是否所有活跃玩家都已匹配当前下注
390        for player in active_players:
391            if player.current_bet < self._game_state.current_bet:
392                return False
393                
394        return True
395        
396    def _advance_to_next_phase(self) -> None:
397        """推进到下一个游戏阶段."""
398        # 收集当前轮的下注到底池
399        self._game_state.collect_bets_to_pot()
400        
401        # 推进阶段
402        if self._game_state.phase == Phase.PRE_FLOP:
403            self._game_state.advance_phase()  # 到FLOP
404            self._game_state.deal_community_cards(3)
405        elif self._game_state.phase == Phase.FLOP:
406            self._game_state.advance_phase()  # 到TURN
407            self._game_state.deal_community_cards(1)
408        elif self._game_state.phase == Phase.TURN:
409            self._game_state.advance_phase()  # 到RIVER
410            self._game_state.deal_community_cards(1)
411        elif self._game_state.phase == Phase.RIVER:
412            self._game_state.advance_phase()  # 到SHOWDOWN
413            
414        # 重置下注轮并设置第一个行动玩家
415        if self._game_state.phase != Phase.SHOWDOWN:
416            self._game_state.reset_betting_round()
417            self._set_first_player_for_new_round()
418            
419    def _set_first_player_for_new_round(self) -> None:
420        """为新的下注轮设置第一个行动玩家."""
421        # 从庄家后面开始找第一个活跃玩家
422        players = self._game_state.players
423        dealer_pos = self._game_state.dealer_position
424        
425        for i in range(1, len(players) + 1):
426            pos = (dealer_pos + i) % len(players)
427            player = players[pos]
428            if player.status == SeatStatus.ACTIVE:
429                self._game_state.current_player = pos
430                return
431                
432        self._game_state.current_player = None
